ADOS-2 Clinical Scoring Implementation Scoring Algorithm pythondef _score_a1_eye_contact(self, values): """Score A1: Eye Contact with clinical interpretation""" gaze_pct, episodes = values

if gaze_pct >= 60 and episodes >= 20: return 0, "Eye contact patterns within typical range", "Good eye contact quality" elif gaze_pct >= 40 or episodes >= 15: return 1, "Mildly reduced eye contact", "Some reduction in initiation" else: return 2, "Markedly limited eye contact", "Significant difficulties"
