def chmod_recursive(path, mode):
    for root, dirs, files in os.walk(path):
        for dir in dirs:
            os.chmod(os.path.join(root, dir), mode)
        for file in files:
            os.chmod(os.path.join(root, file), mode)
    os.chmod(path, mode)

def extract_policy_info(base_results_directory, base_data_directory):
    policy_info = {'results': {}, 'data': {}}

    for directory_type, directory in [('results', base_results_directory), ('data', base_data_directory)]:
        logging.info(f"Extracting policy info from {directory_type} directory: {directory}")

        try:
            for file in os.listdir(directory):
                if file.startswith('top_') and file.endswith('.csv'):
                    logging.debug(f"Processing file in {directory_type} directory: {file}")
                    parts = file.split('_')
                    if len(parts) > 4:
                        field = parts[2]
                        policy_type = parts[3]
                        policy_name = '_'.join(parts[4:]).replace('.csv', '')

                        if field not in policy_info[directory_type]:
                            policy_info[directory_type][field] = {}
                        if policy_type not in policy_info[directory_type][field]:
                            policy_info[directory_type][field][policy_type] = set()

                        policy_info[directory_type][field][policy_type].add(policy_name)
                        logging.debug(f"Added policy in {directory_type}: field={field}, type={policy_type}, name={policy_name}")
                else:
                    logging.debug(f"Skipping file in {directory_type} directory: {file}")

            if not policy_info[directory_type]:
                logging.warning(f"No policy information extracted from {directory_type} directory. Check file naming convention and directory contents.")

        except Exception as e:
            logging.error(f"Error processing {directory_type} directory: {str(e)}")

    return policy_info

def extract_date_range_and_case_count(base_data_directory):
    csv_files = glob.glob(os.path.join(base_data_directory, 'split_files', '*.csv'))

    earliest_date = None
    latest_date = None
    total_cases = 0

    for file in csv_files:
        df = pd.read_csv(file)
        total_cases += len(df)

        if 'item_moderation_date' in df.columns:
            # Convert to datetime, assuming YYYYMMDD format
            df['item_moderation_date'] = pd.to_datetime(df['item_moderation_date'], format='%Y%m%d', errors='coerce')

            file_earliest = df['item_moderation_date'].min()
            file_latest = df['item_moderation_date'].max()

            if pd.notna(file_earliest) and (earliest_date is None or file_earliest < earliest_date):
                earliest_date = file_earliest
            if pd.notna(file_latest) and (latest_date is None or file_latest > latest_date):
                latest_date = file_latest

    if earliest_date and latest_date:
        date_range = f"{earliest_date.strftime('%Y%m%d')} to {latest_date.strftime('%Y%m%d')}"
    else:
        date_range = "N/A"

    return date_range, total_cases

def generate_ngrams(tokens, n):
    if not isinstance(tokens, list):
        return []
    ngrams = [' '.join(tokens[i:i+n]) for i in range(len(tokens)-n+1)]

    # Debug logging
    if any(token.isdigit() for token in tokens):
        logging.debug(f"Numeric tokens in n-gram generation: {tokens}")
        logging.debug(f"Generated {n}-grams: {ngrams}")

    return ngrams

def generate_word_variations(word):
    word = word.lower()
    patterns = [
        rf'{word}+',
        rf'.*{word}.*',
        rf'({word})\1+',
    ]
    return [re.compile(pattern) for pattern in patterns]

# Updated Text preprocessing function
def preprocess_text(text, stop_patterns):
    if pd.isna(text):
        return []

    # Use regex to tokenize, keeping hashtags, numbers, and emojis intact
    tokens = regex.findall(r'\#[\w\d]+|\p{Emoji}|\d+(?:\.\d+)?|\w+', str(text).lower(), regex.UNICODE)

    # Combine consecutive number tokens
    combined_tokens = []
    current_number = ''
    for token in tokens:
        if token.isdigit():
            current_number += token
        else:
            if current_number:
                combined_tokens.append(current_number)
                current_number = ''
            combined_tokens.append(token)
    if current_number:
        combined_tokens.append(current_number)

    return [
        word for word in combined_tokens
        if (word.isalnum() or word.startswith('#') or regex.match(r'\p{Emoji}', word, regex.UNICODE) or word.replace('.','',1).isdigit()) and
        not any(pattern.fullmatch(word.lstrip('#')) for pattern in stop_patterns)
    ]

def dict_to_json(d):
    """
    Convert a dictionary to a JSON string.

    This function handles the case where dictionary keys might not be strings
    (which is a requirement for JSON), by converting all keys to strings.

    Args:
    d (dict): The dictionary to convert

    Returns:
    str: A JSON-formatted string representation of the dictionary
    """
    try:
        # First, try to directly convert to JSON
        return json.dumps(d)
    except TypeError:
        # If that fails, it's likely because of non-string dictionary keys
        # Convert all keys to strings and try again
        return json.dumps({str(k): v for k, v in d.items()})


def load_chunks(folder, start=0, end=None):
    """
    Load processed chunks from disk.

    :param folder: Folder containing the saved chunks
    :param start: Starting chunk index (inclusive)
    :param end: Ending chunk index (exclusive). If None, load all chunks from start.
    :return: DataFrame containing the loaded chunks
    """
    chunk_files = sorted([f for f in os.listdir(folder) if f.startswith('processed_') and f.endswith('.parquet')])

    if end is None:
        end = len(chunk_files)

    chunks = []
    for i in range(start, end):
        chunk = pd.read_parquet(os.path.join(folder, chunk_files[i]))
        chunks.append(chunk)

    return pd.concat(chunks, ignore_index=True)

def split_csv(input_file, output_dir, split_column, chunk_size=50000):
    os.makedirs(output_dir, exist_ok=True)

    file_handles = {}

    def sanitize_filename(name):
        # Convert to lowercase
        name = name.lower()
        # Replace spaces with underscores
        name = name.replace(' ', '_')
        # Remove any non-alphanumeric characters (except underscores)
        name = re.sub(r'[^\w_]', '', name)
        return name

    for chunk in tqdm(pd.read_csv(input_file, chunksize=chunk_size), desc="Splitting CSV"):
        for value, group in chunk.groupby(split_column):
            safe_value = sanitize_filename(str(value))
            if safe_value not in file_handles:
                filename = os.path.join(output_dir, f"{safe_value}.csv")
                os.makedirs(os.path.dirname(filename), exist_ok=True)
                file_handles[safe_value] = open(filename, 'w')
                group.to_csv(file_handles[safe_value], index=False)
            else:
                group.to_csv(file_handles[safe_value], index=False, header=False)

    for handle in file_handles.values():
        handle.close()

    logging.info(f"CSV file split into {len(file_handles)} files based on {split_column}")

def process_caption_text(text):
    if not isinstance(text, str):
        return {
            'original_text': str(text),
            'processed_text': str(text),
            'emoji_data': []
        }

    emoji_data = []
    for char in text:
        if char in emoji.EMOJI_DATA:
            emoji_data.append({
                'emoji': char,
                'description': emoji.demojize(char),
                'count': 1
            })
    return {
        'original_text': text,
        'processed_text': emoji.demojize(text),
        'emoji_data': emoji_data
    }

def generate_co_occurrences(tokens, window_size=5, top_n=100, min_freq=1):
    if not isinstance(tokens, list):
        return {n: {} for n in range(2, 6)}

    co_occurrences = {n: defaultdict(int) for n in range(2, 6)}

    # Increase window size for caption text
    if len(tokens) > 10:  # Assuming longer text is caption
        window_size = 8

    for i in range(len(tokens)):
        for j in range(i+1, min(i+window_size, len(tokens))):
            for n in range(2, min(j-i+2, 6)):
                ngram = tuple(sorted(tokens[i:i+n]))
                co_occurrences[n][ngram] += 1

    # Filter and sort co-occurrences
    for n in range(2, 6):
        co_occurrences[n] = dict(sorted(
            [(k, v) for k, v in co_occurrences[n].items() if v >= min_freq],
            key=lambda x: x[1],
            reverse=True
        )[:top_n])

    return co_occurrences


@timeout(180)  # 3 minutes timeout
def process_chunk(chunk, filename, total_chunks, results_dir):
    chunk_index = chunk.index[0] // len(chunk)
    logging.info(f"Processing chunk {chunk_index + 1}/{total_chunks} of {filename}")

    for field in ['search_term', 'caption_text']:
        source_column = 'unique_search_term' if field == 'search_term' else 'item_caption_text'
        if source_column not in chunk.columns:
            logging.warning(f"Column '{source_column}' not found in the data. Available columns: {chunk.columns.tolist()}")
            continue

        if field == 'caption_text':
            chunk[f'{field}_emoji_processed'] = chunk[source_column].apply(process_caption_text)
            chunk[f'processed_{field}'] = chunk[f'{field}_emoji_processed'].apply(lambda x: preprocess_text(x.get('processed_text', ''), stop_patterns))
        else:
            chunk[f'processed_{field}'] = chunk[source_column].apply(lambda x: preprocess_text(x, stop_patterns) if isinstance(x, str) else [])

        logging.info(f"Sample processed {field}: {chunk[f'processed_{field}'].head().tolist()}")

        co_occurrences = chunk[f'processed_{field}'].apply(
            lambda x: generate_co_occurrences(x,
                                              window_size=5 if field == 'caption_text' else 2,
                                              top_n=100 if field == 'caption_text' else 50,
                                              min_freq=2 if field == 'caption_text' else 1)
        )

        for n in range(2, 6):
            chunk[f'{field}_co_occurrences_{n}'] = co_occurrences.apply(lambda x: json.dumps({json.dumps(k): v for k, v in x[n].items()}))

        for n in [1, 2, 3, 4, 5]:
            chunk[f'{field}_{n}grams'] = chunk[f'processed_{field}'].apply(lambda x: json.dumps(generate_ngrams(x, n)))

            # Debug logging
            if n == 1:  # Check unigrams
                sample = chunk[f'{field}_{n}grams'].head()
                logging.debug(f"Sample {n}-grams for {field}: {sample.tolist()}")

        logging.info(f"Sample 2-grams for {field}: {chunk[f'{field}_2grams'].head().tolist()}")

    save_intermediate_results(chunk, results_dir, filename, chunk_index)

    return chunk_index

            return 0, 0

    file_index, chunk_index = load_checkpoint()

    for file_index, filename in enumerate(tqdm(os.listdir(input_dir), total=total_files, initial=file_index), file_index):
        if filename.endswith('.csv'):
            file_path = os.path.join(input_dir, filename)
            base_filename = os.path.splitext(filename)[0]

            existing_chunks = [f for f in os.listdir(results_dir) if f.startswith(f"processed_{base_filename}_chunk_")]

            if existing_chunks:
                logging.info(f"Skipping file {file_index + 1}/{total_files}: {filename} - processed chunks already exist")
                print(f"Skipping file {file_index + 1}/{total_files}: {filename} - processed chunks already exist")
                continue

            logging.info(f"Processing file {file_index + 1}/{total_files}: {filename}")
            print(f"Processing file {file_index + 1}/{total_files}: {filename}")

            try:
                total_rows = sum(1 for _ in open(file_path)) - 1  # subtract 1 for header
                total_chunks = (total_rows + chunk_size - 1) // chunk_size

                with multiprocessing.Pool(processes=multiprocessing.cpu_count()) as pool:
                    process_chunk_partial = partial(process_chunk, filename=filename, total_chunks=total_chunks, results_dir=results_dir)
                    chunk_reader = pd.read_csv(file_path, chunksize=chunk_size)

                    for processed_chunk_index in tqdm(pool.imap(process_chunk_partial, chunk_reader),
                                                      total=total_chunks-chunk_index,
                                                      initial=chunk_index,
                                                      desc=f"Processing {filename}"):
                        save_checkpoint(file_index, processed_chunk_index + 1)

                        if processed_chunk_index % 10 == 0:  # Collect garbage every 10 chunks
                            gc.collect()

                logging.info(f"Processed {filename}")
                print(f"Processed {filename}")

            except Exception as e:
                logging.error(f"Error processing {filename}: {str(e)}")
                print(f"Error processing {filename}: {str(e)}")

            gc.collect()

    if trimmed_counts:
        logging.info("Summary of trimmed strings:")
        print("Summary of trimmed strings:")
        for column, count in trimmed_counts.items():
            logging.info(f"  {column}: {count} strings trimmed")
            print(f"  {column}: {count} strings trimmed")
    else:
        logging.info("No strings were trimmed during processing")
        print("No strings were trimmed during processing")

def trim_string(s, max_length=100000, column_name=None):  # 100,000 characters
    global trimmed_counts
    if isinstance(s, str) and len(s) > max_length:
        trimmed = s[:max_length] + "... [truncated]"
        if column_name:
            logging.warning(f"String in column '{column_name}' trimmed from {len(s)} to {max_length} characters")
            print(f"String in column '{column_name}' trimmed from {len(s)} to {max_length} characters")
            trimmed_counts[column_name] += 1
        return trimmed
    return s

def save_intermediate_results(chunk, results_dir, original_filename, chunk_index):
    base_filename = os.path.splitext(original_filename)[0]
    chunk_filename = f"processed_{base_filename}_chunk_{chunk_index}_{datetime.now().strftime('%Y%m%d%H%M%S')}.parquet"
    chunk_path = os.path.join(results_dir, chunk_filename)

    for column in chunk.columns:
        if chunk[column].dtype == 'object':
            if column == 'caption_text_emoji_processed':
                chunk[column] = chunk[column].apply(json.dumps)
            else:
                chunk[column] = chunk[column].apply(lambda x: json.dumps(x) if isinstance(x, (list, dict)) else x)
            chunk[column] = chunk[column].apply(lambda x: trim_string(x, column_name=column))

            # Check for numeric content in object columns
            sample = chunk[column].head()
            if any(isinstance(val, (int, float)) or (isinstance(val, str) and val.isdigit()) for val in sample):
                logging.warning(f"Column {column} contains numeric data but is of type object")

    fields = [pa.field(column, pa.large_string()) if dtype == 'object' else
              pa.field(column, pa.int64()) if dtype == 'int64' else
              pa.field(column, pa.float64()) if dtype == 'float64' else
              pa.field(column, pa.large_string())
              for column, dtype in chunk.dtypes.items()]

    schema = pa.schema(fields)
    table = pa.Table.from_pandas(chunk, schema=schema)
    pq.write_table(table, chunk_path, compression='zstd', compression_level=3)

    logging.info(f"Saved intermediate results: {chunk_filename}")
    print(f"Saved intermediate results: {chunk_filename}")

def generate_interpretive_statement(field, n, patterns_found):
    if patterns_found:
        return f"Analysis of {n}-word co-occurrences for {field} was successful. Patterns were found and analyzed."
    else:
        return f"No patterns found for {n}-word co-occurrences in {field}. This could be due to the nature of the text (e.g., short search terms for higher n-grams) or the rarity of such patterns in the data."

def analyze_co_occurrences(folder, field, top_n=100, policy_filter=None):
    co_occurrences = Counter()
    chunk_files = sorted([f for f in os.listdir(folder) if f.startswith('processed_') and f.endswith('.parquet')])

    for chunk_file in tqdm(chunk_files, desc=f"Analyzing co-occurrences for {field}"):
        chunk = pd.read_parquet(os.path.join(folder, chunk_file))
        if policy_filter:
            policy_field, policy_value = policy_filter
            chunk = chunk[chunk[policy_field] == policy_value]

        for n in range(2, 6):
            column_name = f'{field}_co_occurrences_{n}'
            if column_name in chunk.columns:
                chunk_co_occurrences = chunk[column_name].apply(json.loads)
                for co_occ_dict in chunk_co_occurrences:
                    co_occurrences.update({tuple(json.loads(k)): v for k, v in co_occ_dict.items()})

    return pd.DataFrame(sorted(co_occurrences.items(), key=lambda x: x[1], reverse=True)[:top_n],
                        columns=['co_occurrence', 'frequency'])

# Word pattern analysis
def analyze_word_patterns(folder, column, pattern_type, n=1, top_n=100, policy_filter=None):
    pattern_freq = Counter()
    chunk_files = sorted([f for f in os.listdir(folder) if f.startswith('processed_') and f.endswith('.parquet')])

    for chunk_file in tqdm(chunk_files, desc=f"Analyzing {pattern_type} in {column}"):
        chunk = pd.read_parquet(os.path.join(folder, chunk_file), engine='fastparquet')
        if policy_filter:
            policy_field, policy_value = policy_filter
            chunk = chunk[chunk[policy_field] == policy_value]

        column_name = column

        if column_name not in chunk.columns:
            print(f"Warning: Column '{column_name}' not found in {chunk_file}")
            continue

        patterns = chunk[column_name].apply(json.loads)

        for pattern_dict in patterns:
            if isinstance(pattern_dict, (dict, list)):
                if pattern_type == 'co_occurrences':
                    pattern_freq.update({' '.join(k.split('|')): v for k, v in pattern_dict.items()})
                else:
                    pattern_freq.update(pattern_dict)
            else:
                print(f"Unexpected data type in {column_name}: {type(pattern_dict)}")

    if not pattern_freq:
        # Return an empty DataFrame instead of None
        return pd.DataFrame(columns=[pattern_type, 'frequency'])

    return pd.DataFrame(pattern_freq.most_common(top_n), columns=[pattern_type, 'frequency'])

def interpret_word_patterns(df, field, pattern_type):
    if df.empty:
        return f"No {pattern_type} patterns found for {field}."

    top_patterns = df.head(5)
    pattern_column = df.columns[0]
    frequency_column = df.columns[1]

    interpretation = f"The most frequent {pattern_type} in {field} is '{top_patterns.iloc[0][pattern_column]}', occurring {top_patterns.iloc[0][frequency_column]} times. "
    interpretation += f"Other frequent {pattern_type} include: "
    for i in range(1, min(5, len(top_patterns))):
        interpretation += f"'{top_patterns.iloc[i][pattern_column]}' ({top_patterns.iloc[i][frequency_column]} occurrences), "

    interpretation = interpretation.rstrip(', ') + ". "
    interpretation += f"These {pattern_type} provide insight into the most common terms and phrases associated with {field}."

    return interpretation

def interpret_overall_co_occurrences(co_occurrences):
    if co_occurrences.empty:
        return "No co-occurrences found."

    pattern_column = co_occurrences.columns[0]
    frequency_column = co_occurrences.columns[1]

    interpretation = f"""
    The co-occurrence analysis reveals patterns in how terms appear together in the dataset:

    1. Most frequent co-occurrence: The pair "{co_occurrences.iloc[0][pattern_column]}" appears {co_occurrences.iloc[0][frequency_column]} times, suggesting a strong association between these terms in the context of content moderation.

    2. Diversity of co-occurrences: We observed {len(co_occurrences)} distinct co-occurrences, indicating a wide range of term combinations in the dataset.

    3. Frequency distribution: The top 5 co-occurrences account for {co_occurrences.head(5)[frequency_column].sum()} appearances, which represents {co_occurrences.head(5)[frequency_column].sum() / co_occurrences[frequency_column].sum() * 100:.2f}% of all observed co-occurrences.

    4. Contextual insights: The presence of co-occurrences like "{co_occurrences.iloc[1][pattern_column]}" and "{co_occurrences.iloc[2][pattern_column]}" suggests themes or topics that frequently appear together in the content being moderated.

    5. Potential focus areas: Based on these co-occurrences, content moderation efforts might benefit from paying special attention to content containing combinations like "{co_occurrences.iloc[0][pattern_column]}" and "{co_occurrences.iloc[1][pattern_column]}".
    """
    return interpretation

def interpret_ngram_co_occurrences(co_occurrences_by_n):
    if not co_occurrences_by_n or all(df.empty for df in co_occurrences_by_n.values()):
        return "No co-occurrences found across n-gram sizes."

    interpretation = "Analyzing co-occurrences across different n-gram sizes provides insights into the complexity of term associations:\n\n"

    for n in range(2, 6):
        if n in co_occurrences_by_n and not co_occurrences_by_n[n].empty:
            pattern_column = co_occurrences_by_n[n].columns[0]
            frequency_column = co_occurrences_by_n[n].columns[1]

            interpretation += f"{n}-word co-occurrences: The most frequent is \"{co_occurrences_by_n[n].iloc[0][pattern_column]}\", "
            interpretation += f"appearing {co_occurrences_by_n[n].iloc[0][frequency_column]} times. "
            interpretation += f"We observed {len(co_occurrences_by_n[n])} distinct patterns.\n\n"

    interpretation += "Complexity progression: As we move from 2-word to 5-word co-occurrences, "
    interpretation += f"we observe {', '.join([f'{len(co_occurrences_by_n[i])} distinct patterns' for i in range(2, 6) if i in co_occurrences_by_n])} respectively, "
    interpretation += "showing how the complexity and specificity of term associations increase with n-gram size.\n\n"

    interpretation += "Moderation implications: While shorter co-occurrences may be more frequent, "
    interpretation += "longer co-occurrences might provide more context and specificity for targeted content moderation efforts."

    return interpretation

                    return clean_text
            return ""

def decision_process(prompt, valid_responses=None, default=None, allow_multiple_answers=False, allow_numbers=False):
    if valid_responses is None:
        valid_responses = ["yes", "no", "all"] # Default valid responses

    while True:
        response = input(f"{prompt} [default: {default}] " if default else prompt).strip().lower()

        if response == '':
            return default if default is not None else None
        if response in valid_responses:
            return response
        if allow_multiple_answers and ',' in response:
            try:
                responses = [int(x.strip()) for x in response.split(',')]
                return responses
            except ValueError:
                print("Invalid input, please enter numbers separated by commas.")
                continue
        if allow_numbers and response.isdigit():
            return int(response)
        print("Invalid input. Please try again.")

def validate_directory(prompt, default=None):

    if directory_path.lower() == 'exit':

    if os.path.isdir(directory_path):
        print(f"Using directory: {directory_path}")

    else:

        return validate_directory(prompt, default)

def read_custom_stoplist(file_path):
    """Read and return a custom stoplist from a file."""
    try:
        with open(file_path, 'r', encoding='utf-8') as file:
            stopwords = [line.strip() for line in file.readlines() if line.strip()]
        return stopwords
    except FileNotFoundError:
        print(f"File not found: {file_path}")
    except Exception as e:
        print(f"Failed to read custom stoplist: {str(e)}")
    return []

def get_custom_lists(column_name, default_directory=None):
    """Interactively get custom stoplist files from a directory validated by the user."""
    console.print(f"Current working directory: [bold]{os.getcwd()}[/bold]")

    # If no default_directory is provided, use the current working directory
    directory_path = default_directory or os.getcwd()
    if not os.path.exists(directory_path):
        console.print("[red]The default directory does not exist. Please specify the directory.[/red]")
        directory_path = input("Enter the path to the directory: ").strip()

    if not os.path.isdir(directory_path):
        console.print("[red]The path provided is not a directory. Exiting the script.[/red]")
        raise Exception(f"No such file or directory: {directory_path}")

    txt_files = [f for f in os.listdir(directory_path) if f.endswith(".txt")]
    if not txt_files:
        console.print("[red]No .txt files found in the directory.[/red]")
        return []

    console.print("[bold]Available text files:[/bold]")
    for idx, file in enumerate(txt_files):
        console.print(f"{idx + 1}. {file}")

    file_indices = decision_process("Select the numbers of the files you want to use, separated by commas, or type 'all' to use all files:", allow_multiple_answers=True, default="all")
    if file_indices == "all":
        selected_files = txt_files
    else:
        selected_files = [txt_files[int(index) - 1] for index in file_indices.split(',')]

    combined_stopwords = []
    for file in selected_files:
        file_path = os.path.join(directory_path, file)
        if os.path.exists(file_path):
            stopwords = read_custom_stoplist(file_path)
            combined_stopwords.extend(stopwords)
        else:
            console.print(f"[red]File not found: {file_path} - It will be skipped.[/red]")

    return combined_stopwords

def modify_settings(defaults, nltk=False):
    """Modify settings based on user input, dynamically handling different data types."""

            'min_freq': int,
            'ngrams': list, # List of integers

            'keywords': str,
            'keywords_logic': str,
            'apply_english_stoplist': bool,
            'custom_stoplist': list # This could be a list of strings (stopwords)
        }
    else:
        # Default expected types if not specifically for NLTK
        expected_types = {key: type(value) for key, value in defaults.items()}

    for key, current_value in defaults.items():
        expected_type = expected_types.get(key, type(current_value))
        prompt = f"Enter new value for {key} ({current_value}) or press [Enter] to keep the current value:"
        response = decision_process(prompt, default=str(current_value))

        if response != "":
            try:
                # Convert list from string input correctly
                if expected_type == list:
                    if key == 'ngrams': # Expecting list of integers
                        defaults[key] = [int(item.strip()) for item in response.split(',')]
                    else:
                        defaults[key] = response.split(',')
                elif expected_type == bool:
                    defaults[key] = response.lower() in ['true', 'yes', '1']
                else:
                    defaults[key] = expected_type(response)
            except ValueError:
                console.print(f"[red]Invalid input for {key}. Expected {expected_type.__name__}.[/red]")
            else:
                console.print(f"Updated {key} to {response}")

    return defaults

def auto_fetch_uois(table_name, sql_condition):
    """Automatically fetch UIDs based on distinct entries from a specified table."""
    query = f"SELECT DISTINCT user_id FROM {table_name}" + (f" WHERE {sql_condition}" if sql_condition else "")
    unique_uids = spark.sql(query).collect()
    uois = [{
        'uoi_name': f'UOI_{index + 1}',
        'uids': [uid.user_id],
        'table_name': table_name,
        'sql_condition': sql_condition
    } for index, uid in enumerate(unique_uids)]
    return uois

def fetch_and_process_data(uoi):
    """Process data for a given UOI."""
    uoi_name, uids, table_name, sql_condition = uoi['uoi_name'], uoi['uids'], uoi['table_name'], uoi['sql_condition']
    ids = ", ".join(str(uid) for uid in uids)
    file_path = os.path.join(base_data_directory, f"{uoi_name}_consolidated.csv")

    if not os.path.exists(file_path):
        query = f"""
        SELECT
            user_id,
            view_source,
            item_id,
            text,
            sticker_text_info,
            event_time,
            first(policy_map) as policy_map,
            MAX(partition_date) OVER (PARTITION BY user_id, item_id) as last_date,
            MIN(partition_date) OVER (PARTITION BY user_id, item_id) as first_date
        FROM (
            SELECT *,
                   ROW_NUMBER() OVER (PARTITION BY user_id, item_id ORDER BY event_time DESC) as rn
            FROM {table_name}
            WHERE user_id IN ({ids}) AND {sql_condition}
        ) sub
        WHERE sub.rn = 1
        GROUP BY user_id, view_source, item_id, text, sticker_text_info, event_time, partition_date
        """
        df = spark.sql(query)
        df = df.withColumn("policy_names_string", concat_ws(', ', expr("map_keys(policy_map)")))
        df = df.withColumn("clean_sticker_text", clean_sticker_text_udf(col("sticker_text_info")))
        df = df.withColumn("text_combined", concat_ws(' ', df.text, df.clean_sticker_text))
        df.show(5)
        df.toPandas().to_csv(file_path, index=False)

def generate_segmented_csvs(base_data_directory, uois):
    import os
    import pandas as pd
    from tqdm import tqdm

    for uoi in tqdm(uois, desc="Generating segmented CSVs", unit="UOI"):
        data_path = os.path.join(base_data_directory, f"{uoi['uoi_name']}_consolidated.csv")
        print(f"Attempting to access file: {data_path}") # Debug line
        if os.path.exists(data_path):
            df = pd.read_csv(data_path)
            df['event_time'] = pd.to_datetime(df['event_time'], errors='coerce') # Correct handling of NaT values
            df = df.dropna(subset=['event_time']) # Ensuring no NaT values before operations

            if not df.empty:
                total_duration = df['event_time'].max() - df['event_time'].min()
                first_tertile_end = df['event_time'].min() + total_duration / 3
                second_tertile_end = df['event_time'].min() + total_duration * 2 / 3

                early_tertile = df[df['event_time'] <= first_tertile_end]
                mid_tertile = df[(df['event_time'] > first_tertile_end) & (df['event_time'] <= second_tertile_end)]
                late_tertile = df[df['event_time'] > second_tertile_end]

                n = len(df)
                first_25 = df.iloc[:n//4]
                last_25 = df.iloc[-n//4:]

                analyze_and_save_segment(df, 'full', base_data_directory, uoi['uoi_name'])
                analyze_and_save_segment(first_25, 'first_25', base_data_directory, uoi['uoi_name'])
                analyze_and_save_segment(last_25, 'last_25', base_data_directory, uoi['uoi_name'])
                analyze_and_save_segment(early_tertile, 'early_tertile', base_data_directory, uoi['uoi_name'])
                analyze_and_save_segment(mid_tertile, 'mid_tertile', base_data_directory, uoi['uoi_name'])
                analyze_and_save_segment(late_tertile, 'late_tertile', base_data_directory, uoi['uoi_name'])
            else:
                print(f"No valid data to process for {uoi['uoi_name']}.")
        else:
            print(f"File not found for {uoi['uoi_name']}")

def generate_summary_csv(base_data_directory, uois):
    """Generates a comprehensive summary CSV with detailed metrics for each UOI."""
    import os
    import pandas as pd

    summary = []
    for uoi in uois:
        data_path = os.path.join(base_data_directory, f"{uoi['uoi_name']}_consolidated.csv")
        print(f"Checking for data file at: {data_path}") # Debugging line
        if os.path.exists(data_path):
            df = pd.read_csv(data_path)
            df['policy_present'] = df['policy_map'].notna()
            df_sorted = df.sort_values(by='event_time')
            first_25 = df_sorted.iloc[:len(df_sorted)//4]
            last_25 = df_sorted.iloc[3*len(df_sorted)//4:]
            first_tertile = df_sorted.iloc[:len(df_sorted)//3]
            mid_tertile = df_sorted.iloc[len(df_sorted)//3:2*len(df_sorted)//3]
            last_tertile = df_sorted.iloc[2*len(df_sorted)//3:]

            def calculate_percentage(data):
                return 100 * data['policy_present'].sum() / len(data) if len(data) > 0 else 0

            summary_record = {
                'UOI': uoi['uoi_name'],
                'policy_perc_total': calculate_percentage(df),
                'policy_perc_first': calculate_percentage(first_25),
                'policy_perc_last': calculate_percentage(last_25),
                'policy_perc_early': calculate_percentage(first_tertile),
                'policy_perc_mid': calculate_percentage(mid_tertile),
                'policy_perc_late': calculate_percentage(last_tertile)
            }
            summary.append(summary_record)
            print(f"Summary data processed for {uoi['uoi_name']}")
        else:
            print(f"No data file found for {uoi['uoi_name']}, skipping.")

    summary_df = pd.DataFrame(summary)
    summary_csv_path = os.path.join(base_data_directory, "summary_uois.csv")
    summary_df.to_csv(summary_csv_path, index=False)
    print(f"Summary CSV generated at {summary_csv_path}.")

def format_emojis(text):
    """
    Convert emojis in the text to a UTF-8 string representation without spaces.
    """
    if pd.isna(text):
        return ''
    else:
        text = str(text)
        formatted_text = emoji.demojize(text, delimiters=("::", "::"))
        formatted_text = re.sub(r"::(\\s+)?([a-zA-Z_]+(\s+[a-zA-Z_]+)+)(\s+)?::", lambda match: "::" + match.group(2).replace(" ", "_") + "::", formatted_text)
        return formatted_text

def deformat_emojis(text):
    """
    Convert emoji string representations back to their graphical form.
    """
    if pd.isna(text):
        return ''
    else:
        return emoji.emojize(str(text), use_aliases=True)

def preprocess_text_for_analysis(text, column_name):
    """
    Preprocess text by formatting emojis and removing newlines.
    """
    text = format_emojis(text.replace("\n", " ")).strip() if pd.notna(text) else text
    return text

def postprocess_text_for_output(df, column_name):
    """
    Postprocess text in a DataFrame column by reformatting emojis to their graphical form.
    """
    df[column_name] = df[column_name].apply(lambda x: deformat_emojis(x) if pd.notna(x) else x)
    return df

def process_nltk_analysis(data_path, settings, column_name, base_results_directory):
    try:
        filename = os.path.basename(data_path)
        uoi_id = re.search(r"UOI_(\d+)", filename).group(0)
        uoi_nltk_directory = os.path.join(base_results_directory, uoi_id)
        os.makedirs(uoi_nltk_directory, exist_ok=True)

        df = pd.read_csv(data_path)
        text = ' '.join(df[column_name].dropna().astype(str))
        tokens = nltk.word_tokenize(text)

        stop_words = set()
        if settings.get('apply_english_stoplist'):
            stop_words.update(nltk.corpus.stopwords.words('english'))
        if settings.get('custom_stoplist'):
            stop_words.update(settings['custom_stoplist'])

        if settings['apply_stoplist_to'] == 'all_words':
            filtered_tokens = [token for token in tokens if token.lower() not in stop_words and token.isalpha()]
        else:
            filtered_tokens = tokens

        ngram_funcs = {
            2: (BigramCollocationFinder.from_words, BigramAssocMeasures()),
            3: (TrigramCollocationFinder.from_words, TrigramAssocMeasures()),
            4: (QuadgramCollocationFinder.from_words, QuadgramAssocMeasures())
        }

        # Handling various n-grams as specified in settings
        for ngram_size in settings.get('ngrams', [2]):
            if ngram_size in ngram_funcs:
                finder_class, measures = ngram_funcs[ngram_size]
                finder = finder_class(filtered_tokens)
                finder.apply_word_filter(lambda w: w.lower() in stop_words or not w.isalpha())
                finder.apply_freq_filter(settings.get('min_freq', 5))

                ngrams = finder.nbest(measures.likelihood_ratio, settings.get('num_ngrams', 100))
                ngram_freqs = {(''.join(ngram)): finder.ngram_fd[ngram] for ngram in ngrams}

                if not ngram_freqs:
                    console.print(f"No {ngram_size}-grams met the frequency threshold for {data_path}.")
                    continue

                output_file = os.path.join(uoi_nltk_directory, f"{filename.replace('.csv', '')}_{column_name}_NLTK_{ngram_size}gram.csv")
                with open(output_file, 'w', newline='', encoding='utf-8') as f:
                    writer = csv.writer(f)
                    writer.writerow(['ngram', 'frequency'])
                    for ngram, freq in ngram_freqs.items():
                        writer.writerow([ngram, freq])

                word_cloud_filename = f"{filename.replace('.csv', '')}_{column_name}_NLTK_{ngram_size}gram_wordcloud.png"
                generate_word_cloud(ngram_freqs, uoi_nltk_directory, word_cloud_filename, f"{ngram_size}-gram Word Cloud", column_name)

    except Exception as e:
        console.print(f"[red]An error occurred while processing {data_path}: {e}[/red]")

        lda.fit(tfidf_matrix)

        topics = {}
        for topic_idx, topic in enumerate(lda.components_):
            topics[f"Topic {topic_idx + 1}"] = [(feature_names[i], topic[i]) for i in topic.argsort()[:-settings['top_words'] - 1:-1]]

        topics_df = pd.DataFrame({key: [val[0] for val in value] for key, value in topics.items()})
        topics_output_file = os.path.join(uoi_nltk_directory, f"{filename.replace('.csv', '')}_{column_name}_NLTK_topics.csv")
        topics_df.to_csv(topics_output_file)

        broad_intro_text = "Keyword topic analysis through NLTK and TF-IDF leverages advanced statistical models to distill coherent themes from large volumes of text, aiding in the understanding of prevalent topics within the dataset."
        print(broad_intro_text)

        narrative = f"NLTK Keyword Topic Analysis for {column_name} ({filename}): This analysis identified {len(topics)} distinct topics, which illustrate the variety and depth of content within the dataset. "
        narrative += "These topics are derived from clusters of terms that commonly appear together, providing a window into the overarching subjects and concerns that permeate the text."
        if len(topics) == 0:
            narrative += " However, no significant topics were detected, indicating potential issues with data sparsity or the need for parameter adjustments."
        print(narrative)

        print(f"NLTK keyword topic analysis completed for {filename}. Results and narrative saved in {uoi_nltk_directory}")
    except Exception as e:
        print(f"Failed to complete NLTK keyword topic analysis for {data_path}: {e}")

def process_nltk_collocation_analysis(data_path, settings, column_name, base_results_directory, data_size):
    try:
        filename = os.path.basename(data_path)
        uoi_id = re.search(r"UOI_(\d+)", filename).group(0)

        os.makedirs(uoi_nltk_directory, exist_ok=True)

        df = pd.read_csv(data_path)
        text = ' '.join(df[column_name].dropna().astype(str))
        tokens = nltk.word_tokenize(text)
        filtered_tokens = [word for word in tokens if word.lower() not in settings['custom_stoplist'] and word.isalpha()]

        finder = BigramCollocationFinder.from_words(filtered_tokens)
        finder.apply_freq_filter(settings['min_freq'])
        bigram_measures = nltk.collocations.BigramAssocMeasures()
        collocations = finder.nbest(bigram_measures.likelihood_ratio, settings['num_collocations'])

        collocations_output_file = os.path.join(uoi_nltk_directory, f"{filename.replace('.csv', '')}_{column_name}_NLTK_collocations.csv")
        with open(collocations_output_file, 'w', newline='', encoding='utf-8') as f:
            writer = csv.writer(f)
            writer.writerow(['Collocation', 'Score'])
            for collocation in collocations:
                writer.writerow([' '.join(collocation), finder.score_ngram(bigram_measures.likelihood_ratio, collocation[0], collocation[1])])

        broad_intro_text = "Collocation analysis through NLTK identifies word pairs that frequently occur together within the text, shedding light on the syntactic and semantic relationships that characterize the language use."
        print(broad_intro_text)

        narrative = f"NLTK Collocation Analysis for {column_name} ({filename}): Analyzed {data_size} text elements. "
        narrative += "This analysis helps reveal common linguistic structures and significant word pairs, offering insights into how terms are contextually bound within the dataset. "
        if len(collocations) > 0:
            top_collocations = ", ".join([' '.join(pair) for pair in collocations[:5]])
            narrative += f"The most significant collocations include: {top_collocations}. These pairs indicate prevalent themes or discussion points."
        else:
            narrative += "No significant collocations were identified, which may suggest either a dispersed use of language or the need for adjusted analysis parameters to better capture meaningful relationships."
        print(narrative)

        print(f"NLTK collocation analysis completed for {filename}. Results and narrative saved in {uoi_nltk_directory}")

def process_sequential_ngram_analysis(data_path, settings, column_name, base_results_directory):

        df = pd.read_csv(data_path)
        text = ' '.join(df[column_name].dropna().astype(str))
        tokens = nltk.word_tokenize(text)

        keywords = settings.get('keywords', [])
        if keywords:
            keyword_ngrams = []
            for ngram_size in range(2, len(keywords) + 1):
                keyword_ngrams.extend(nltk.ngrams(tokens, ngram_size))

            keyword_freq = nltk.FreqDist(keyword_ngrams)
            keyword_freq_filtered = {ngram: freq for ngram, freq in keyword_freq.items() if all(keyword.lower() in ngram for keyword in keywords)}

            output_file = os.path.join(uoi_nltk_directory, f"{filename.replace('.csv', '')}_{column_name}_NLTK_keyword_ngrams.csv")
            with open(output_file, 'w', newline='', encoding='utf-8') as f:
                writer = csv.writer(f)
                writer.writerow(['ngram', 'frequency'])
                for ngram, freq in keyword_freq_filtered.items():
                    writer.writerow([' '.join(ngram), freq])

            console.print(f"Sequential n-gram analysis based on keywords completed for {data_path}, column: {column_name}")
        else:
            console.print(f"No keywords provided for sequential n-gram analysis in {data_path}, column: {column_name}")

    except Exception as e:
        console.print(f"[red]An error occurred during sequential n-gram analysis for {data_path}, column: {column_name}: {e}[/red]")

def find_co_occurrences(texts, window_size, stop_words):
    # Tokenize and filter stopwords
    all_words = []
    for text in texts:
        words = [word.lower() for word in word_tokenize(text) if word.lower() not in stop_words and word.isalpha()]
        all_words.extend(words)

    # Find co-occurrences within the window size
    co_occurrences = []
    for i in range(len(all_words)):
        target_word = all_words[i]
        start = max(0, i - window_size)
        end = min(len(all_words), i + window_size + 1)
        for j in range(start, end):
            if i != j:
                co_occurred_word = all_words[j]
                pair = tuple(sorted([target_word, co_occurred_word]))
                co_occurrences.append(pair)

    # Frequency distribution of co-occurring word pairs
    co_occurrence_freq = FreqDist(co_occurrences)
    return co_occurrence_freq

        # Extract co-occurrences

        with open(co_occurrence_file, 'w', newline='', encoding='utf-8') as f:

def autonomous_survival_analysis(base_data_directory, base_results_directory, time_column, event_column):
    uoi_files = os.listdir(base_data_directory)
    for file in uoi_files:
        if file.endswith(".csv"):
            filepath = os.path.join(base_data_directory, file)
            uoi = file.split('_')[1]
            segment = '_'.join(file.split('_')[2:-1])
            uoi_dir = Path(base_results_directory) / f"UOI_{uoi}"
            uoi_dir.mkdir(parents=True, exist_ok=True)

            df = pd.read_csv(filepath)
            if time_column in df.columns and event_column in df.columns:
                df[time_column] = pd.to_datetime(df[time_column])
                df.sort_values(by=time_column, inplace=True)
                df['duration'] = df[time_column].diff().dt.total_seconds() # Time differences in seconds
                df = df.dropna(subset=['duration'])
                df['event_observed'] = df[event_column].apply(lambda x: 1 if not pd.isna(x) and x.strip() != '' else 0)

                if df['duration'].notna().all() and len(df) > 1:
                    kmf = KaplanMeierFitter()
                    kmf.fit(df['duration'], event_observed=df['event_observed'], label=segment)
                    kmf.plot_survival_function() # Plot survival function for the segment
                    plt.title(f'Survival Function for {segment} of UOI {uoi}')
                    plt.xlabel('Time in Seconds')
                    plt.ylabel('Survival Probability')
                    plt.savefig(uoi_dir / f'{segment}_survival_function.png')
                    plt.close()

                    console.print(f"Survival Analysis for segment '{segment}' of UOI {uoi}:")
                    console.print(f"Analyzed the time until the first violation event. The median survival time is {kmf.median_survival_time_} seconds.")
                    console.print("This analysis helps in understanding the typical duration until a policy violation occurs, providing insights into the effectiveness of content moderation over time.")
                else:
                    console.print("Insufficient data to perform survival analysis for segment '{segment}' of UOI {uoi}.")
            else:
                console.print("Required columns for survival analysis are missing in {segment} of UOI {uoi}.")
                console.print("Ensure the data includes time stamps and event observations to perform this analysis, crucial for evaluating the timing and frequency of critical events.")

def decision_process(prompt, valid_responses=None, default=None, allow_multiple_answers=False, allow_numbers=False):
    if valid_responses is None:
        valid_responses = ["yes", "no", "all"] # Default valid responses

    while True:
        response = input(f"{prompt} [default: {default}] " if default else prompt).strip().lower()

        if response == '':
            return default if default is not None else None
        if response in valid_responses:
            return response
        if allow_multiple_answers and ',' in response:
            try:
                responses = [int(x.strip()) for x in response.split(',')]
                return responses
            except ValueError:
                print("Invalid input, please enter numbers separated by commas.")
                continue
        if allow_numbers and response.isdigit():
            return int(response)
        print("Invalid input. Please try again.")

        # Default expected types if not specifically for NLTK
        expected_types = {key: type(value) for key, value in defaults.items()}

    for key, current_value in defaults.items():
        expected_type = expected_types.get(key, type(current_value))
        prompt = f"Enter new value for {key} ({current_value}) or press [Enter] to keep the current value:"
        response = decision_process(prompt, default=str(current_value))

        if response != "":
            try:
                # Convert list from string input correctly
                if expected_type == list:
                    if key == 'ngrams': # Expecting list of integers
                        defaults[key] = [int(item.strip()) for item in response.split(',')]
                    else:
                        defaults[key] = response.split(',')
                elif expected_type == bool:
                    defaults[key] = response.lower() in ['true', 'yes', '1']
                else:
                    defaults[key] = expected_type(response)
            except ValueError:
                console.print(f"[red]Invalid input for {key}. Expected {expected_type.__name__}.[/red]")
            else:
                console.print(f"Updated {key} to {response}")

    return defaults

def format_emojis(text):
    """
    Convert emojis in the text to a UTF-8 string representation without spaces.
    """
    if pd.isna(text):
        return ''
    else:
        text = str(text)
        formatted_text = emoji.demojize(text, delimiters=("::", "::"))
        formatted_text = re.sub(r"::(\\s+)?([a-zA-Z_]+(\s+[a-zA-Z_]+)+)(\s+)?::", lambda match: "::" + match.group(2).replace(" ", "_") + "::", formatted_text)
        return formatted_text

def deformat_emojis(text):
    """
    Convert emoji string representations back to their graphical form.
    """
    if pd.isna(text):
        return ''
    else:
        return emoji.emojize(str(text))

def preprocess_text_for_analysis(text, column_name):
    """
    Preprocess text by formatting emojis and removing newlines.
    """
    text = format_emojis(text.replace("\n", " ")).strip() if pd.notna(text) else text
    return text

def postprocess_text_for_output(input_data, column_name=None):
    """
    Postprocess text in a DataFrame column or a string by reformatting emojis to their graphical form.
    """
    if isinstance(input_data, pd.DataFrame):
        input_data[column_name] = input_data[column_name].apply(lambda x: deformat_emojis(x) if pd.notna(x) else x)
        return input_data
    elif isinstance(input_data, str):
        return deformat_emojis(input_data)
    else:
        raise TypeError("Input must be a pandas DataFrame or a string")

def initialize_analyzer(lemmatizer_langs, file_type, stoplists, custom_stoplists=[]):

def extract_uoi_id(filename):
    match = re.search(r"UOI_(\d+)", filename)
    if match:
        return match.group(0)
    else:
        logging.warning(f"UOI ID not found in filename: {filename}")
        return None

def perform_frequency_analysis(analyzer, documents, filename, output_path, settings, column_name):

def perform_theme_extraction(analyzer, documents, filename, topics_path, themes_path, settings, column_name):
    try:
        console.print(f"Starting theme extraction for {filename}, column: {column_name}...")
        preprocessed_documents = [preprocess_text_for_analysis(doc, column_name) for doc in documents]
        console.print(f"Preprocessed documents preview (first 3):\n{preprocessed_documents[:3]}")

        # Capture the printed output
        captured_output = io.StringIO()
        sys.stdout = captured_output

        analyzer.run_full_mem(
            texts=preprocessed_documents,
            lemmatize=True,
            minimum_word_count=settings['theme_min_word_count'],
            model_top_n_words=settings['theme_top_n_words'],
            numFactors=settings['theme_numFactors'],
            min_loading=settings['theme_min_loading'],
            print_factors=True,
            verbose=True
        )

        sys.stdout = sys.__stdout__  # Restore the original stdout
        captured_output.seek(0)  # Reset the StringIO buffer to the beginning

        if not hasattr(analyzer, 'loading_matrix') or analyzer.loading_matrix is None:
            console.print(f"[red]No significant themes were extracted for {filename}, column: {column_name}. Skipping export and visualization.[/red]")
            return

        # Process the captured output and save it as a CSV file
        factors_df = pd.DataFrame(columns=['Factor', 'Words'])
        for line in captured_output.readlines():
            if line.startswith('Factor '):
                factor_words = line.strip().split(': ')[1]
                factor_num = re.findall(r'Factor (\d+):', line)[0]
                factors_df = factors_df.append({'Factor': f'Factor {factor_num}', 'Words': factor_words}, ignore_index=True)

        factors_path = os.path.join(os.path.dirname(topics_path), f"{filename[:-4]}_{column_name}_factors.csv")
        factors_df.to_csv(factors_path, index=False)
        console.print(f"Factors (themes) saved to {factors_path}.")

        # The first column is considered to have theme words; process the text in this column to replace text emojis
        if hasattr(analyzer.loading_matrix, 'iloc'):
            analyzer.loading_matrix.iloc[:, 0] = analyzer.loading_matrix.iloc[:, 0].apply(postprocess_text_for_output, column_name=column_name)

        # Now save the dataframe to CSV after reinserting emojis
        analyzer.loading_matrix.to_csv(topics_path)
        console.print(f"Theme extraction completed and data saved to {topics_path}.")

        # Ensure the directory for word clouds exists
        os.makedirs(themes_path, exist_ok=True)
        # Generate and save word clouds
        analyzer.loading_matrix.get_wordclouds(
            min_loading=settings['theme_min_loading'],
            colormap='ocean',
            show_word_clouds=True,
            export_word_clouds=True,
            export_folder=themes_path
        )
        console.print(f"Word clouds stored in {themes_path}.")
        narrative = f"FATE theme extraction for {column_name} in {filename} has uncovered underlying patterns and themes within the text. This analysis helps reveal the dominant topics and concepts, providing a deeper understanding of the content's central ideas."
        details = f"FATE theme extraction performed with a minimum word count of {settings['theme_min_word_count']}, top {settings['theme_top_n_words']} words per theme, {settings['theme_numFactors']} themes, minimum loading of {settings['theme_min_loading']}, and lemmatization enabled."
        return narrative, details

    except np.linalg.LinAlgError as e:
        console.print(f"[red]Theme extraction error due to algebraic issue: {str(e)}[/red]")
        default_narrative = "Theme extraction could not be completed due to an algebraic issue."
        default_details = f"Error: {str(e)}"
        return default_narrative, default_details

    except Exception as e:
        console.print(f"[red]Theme extraction failed for {filename}, column: {column_name}: {e}[/red]")
        logging.warning(f"FATE theme extraction failed for {filename}, column: {column_name}: {e}")
        default_narrative = "Theme extraction failed due to an error."
        default_details = f"Error: {str(e)}"
        return default_narrative, default_details

def process_nltk_analysis(tokens, output_path, settings, ngram_size):
    try:
        stop_words = set()
        if settings.get('apply_english_stoplist'):
            stop_words.update(nltk.corpus.stopwords.words('english'))
        if settings.get('custom_stoplist'):
            stop_words.update(settings['custom_stoplist'])

        if settings.get('apply_stoplist_to', 'all_words') == 'all_words':
            filtered_tokens = [token for token in tokens if token.lower() not in stop_words and token.isalpha()]
        else:
            filtered_tokens = tokens

        ngram_funcs = {
            2: (BigramCollocationFinder.from_words, BigramAssocMeasures()),
            3: (TrigramCollocationFinder.from_words, TrigramAssocMeasures()),
            4: (QuadgramCollocationFinder.from_words, QuadgramAssocMeasures())
        }

        if ngram_size in ngram_funcs:
            finder_class, measures = ngram_funcs[ngram_size]
            finder = finder_class(filtered_tokens)
            finder.apply_word_filter(lambda w: w.lower() in stop_words or not w.isalpha())
            finder.apply_freq_filter(settings.get('min_freq', 5))

            ngrams = finder.nbest(measures.likelihood_ratio, settings.get('num_ngrams', 100))
            ngram_freqs = {(''.join(ngram)): finder.ngram_fd[ngram] for ngram in ngrams}

            if not ngram_freqs:
                console.print(f"No {ngram_size}-grams met the frequency threshold.")
                return f"No {ngram_size}-grams met the frequency threshold.", ""

            with open(output_path, 'w', newline='', encoding='utf-8') as f:
                writer = csv.writer(f)
                writer.writerow(['ngram', 'frequency'])
                for ngram, freq in ngram_freqs.items():
                    writer.writerow([ngram, freq])

            word_cloud_filename = f"{output_path[:-4]}_wordcloud.png"
            narrative, details = generate_word_cloud(ngram_freqs, word_cloud_filename, f"{ngram_size}-gram Word Cloud")
            return narrative, details

        return "No n-grams met the frequency threshold.", ""

    except Exception as e:
        console.print(f"[red]An error occurred while processing NLTK analysis: {e}[/red]")
        return "An error occurred while processing the data.", str(e)

def process_nltk_frequency_analysis(tokens, output_path, settings):
    try:
        stop_words = set(nltk.corpus.stopwords.words('english'))
        if settings.get('apply_english_stoplist', False):
            stop_words.update(settings.get('custom_stoplist', []))

        filtered_tokens = [token for token in tokens if token.lower() not in stop_words and token.isalpha()]

        freq_dist = FreqDist(filtered_tokens)
        freq_dist_df = pd.DataFrame(freq_dist.most_common(settings['num_ngrams']), columns=['Token', 'Frequency'])
        freq_dist_df.to_csv(output_path, index=False)

        broad_intro_text = "NLTK frequency analysis quantifies the occurrence of words within the text, highlighting the most prevalent terms to provide insights into the dominant themes and language use."
        print(broad_intro_text)

        narrative = f"The frequency analysis has processed {len(tokens)} text items, revealing key terms that shape the discourse within the dataset. "
        narrative += "This analysis not only underscores the main topics but also aids in identifying potential areas for in-depth linguistic or thematic exploration."
        if freq_dist_df.empty:
            narrative += " No significant words were identified, suggesting either a broad vocabulary without prevalent terms or the need for different analytical parameters."
        print(narrative)

        print(f"NLTK frequency analysis completed. Results and narrative saved in {output_path}")
        return narrative, "NLTK frequency analysis completed successfully."
    except Exception as e:
        print(f"Failed to complete NLTK frequency analysis: {e}")
        return "NLTK frequency analysis failed.", str(e)

def process_nltk_keyword_topic_analysis(text, output_path, settings):
    try:
        vectorizer = TfidfVectorizer(stop_words=settings['custom_stoplist'], max_df=0.90, min_df=3)
        tfidf_matrix = vectorizer.fit_transform([text])
        feature_names = vectorizer.get_feature_names_out()

        lda = LatentDirichletAllocation(n_components=3, random_state=42)
        lda.fit(tfidf_matrix)

        topics = {}
        for topic_idx, topic in enumerate(lda.components_):
            topics[f"Topic {topic_idx + 1}"] = [(feature_names[i], topic[i]) for i in topic.argsort()[:-settings['top_words'] - 1:-1]]

        topics_df = pd.DataFrame({key: [val[0] for val in value] for key, value in topics.items()})
        topics_df.to_csv(output_path)

        broad_intro_text = "Keyword topic analysis through NLTK and TF-IDF leverages advanced statistical models to distill coherent themes from large volumes of text, aiding in the understanding of prevalent topics within the dataset."
        print(broad_intro_text)

        narrative = f"NLTK Keyword Topic Analysis: This analysis identified {len(topics)} distinct topics, which illustrate the variety and depth of content within the dataset. "
        narrative += "These topics are derived from clusters of terms that commonly appear together, providing a window into the overarching subjects and concerns that permeate the text."
        if len(topics) == 0:
            narrative += " However, no significant topics were detected, indicating potential issues with data sparsity or the need for parameter adjustments."
        print(narrative)

        print(f"NLTK keyword topic analysis completed. Results and narrative saved in {output_path}")
        return narrative, "NLTK keyword topic analysis completed successfully."
    except Exception as e:
        print(f"Failed to complete NLTK keyword topic analysis: {e}")
        return "NLTK keyword topic analysis failed.", str(e)

def process_nltk_collocation_analysis(tokens, output_path, settings):
    try:
        filtered_tokens = [word for word in tokens if word.lower() not in settings['custom_stoplist'] and word.isalpha()]

        finder = BigramCollocationFinder.from_words(filtered_tokens)
        finder.apply_freq_filter(settings['min_freq'])
        bigram_measures = nltk.collocations.BigramAssocMeasures()
        collocations = finder.nbest(bigram_measures.likelihood_ratio, settings['num_collocations'])

        with open(output_path, 'w', newline='', encoding='utf-8') as f:
            writer = csv.writer(f)
            writer.writerow(['Collocation', 'Score'])
            for collocation in collocations:
                writer.writerow([' '.join(collocation), finder.score_ngram(bigram_measures.likelihood_ratio, collocation[0], collocation[1])])

        broad_intro_text = "Collocation analysis through NLTK identifies word pairs that frequently occur together within the text, shedding light on the syntactic and semantic relationships that characterize the language use."
        print(broad_intro_text)

        narrative = f"NLTK Collocation Analysis: Analyzed {len(tokens)} text elements. "
        narrative += "This analysis helps reveal common linguistic structures and significant word pairs, offering insights into how terms are contextually bound within the dataset. "
        if len(collocations) > 0:
            top_collocations = ", ".join([' '.join(pair) for pair in collocations[:5]])
            narrative += f"The most significant collocations include: {top_collocations}. These pairs indicate prevalent themes or discussion points."
        else:
            narrative += "No significant collocations were identified, which may suggest either a dispersed use of language or the need for adjusted analysis parameters to better capture meaningful relationships."
        print(narrative)

        print(f"NLTK collocation analysis completed. Results and narrative saved in {output_path}")
        return narrative, "NLTK collocation analysis completed successfully."
    except Exception as e:
        print(f"Failed to complete NLTK collocation analysis: {e}")
        return "NLTK collocation analysis failed. This failure may affect our understanding of linguistic patterns and contextual relationships within the text.", str(e)

def process_sequential_ngram_analysis(tokens, output_path, settings):
    try:
        keywords = settings.get('keywords', [])
        if keywords:
            keyword_ngrams = []
            for ngram_size in range(2, len(keywords) + 1):
                keyword_ngrams.extend(nltk.ngrams(tokens, ngram_size))

            keyword_freq = nltk.FreqDist(keyword_ngrams)
            keyword_freq_filtered = {ngram: freq for ngram, freq in keyword_freq.items() if all(keyword.lower() in ngram for keyword in keywords)}

            with open(output_path, 'w', newline='', encoding='utf-8') as f:
                writer = csv.writer(f)
                writer.writerow(['ngram', 'frequency'])
                for ngram, freq in keyword_freq_filtered.items():
                    writer.writerow([' '.join(ngram), freq])

            console.print(f"Sequential n-gram analysis based on keywords completed.")
            narrative = f"Sequential n-gram analysis has identified ngrams containing the specified keywords, providing insights into the context and usage of these terms within the text."
            details = f"Sequential n-gram analysis performed using keywords: {', '.join(keywords)}. Results saved in {output_path}."
            return narrative, details
        else:
            console.print(f"No keywords provided for sequential n-gram analysis.")
            return "No keywords provided for sequential n-gram analysis.", ""

    except Exception as e:
        console.print(f"[red]An error occurred during sequential n-gram analysis: {e}[/red]")
        logging.warning(f"An error occurred during sequential n-gram analysis: {e}")
        return "An error occurred during sequential n-gram analysis.", str(e)

def comparative_survival_analysis(dfs, time_column, output_path):
    try:
        survival_fig, ax = plt.subplots(figsize=(12, 8))

        detailed_narrative = {}

        for filename, df in dfs.items():
            segment = extract_segment_from_filename(filename)

            if time_column in df.columns:
                df[time_column] = pd.to_datetime(df[time_column])
                df.sort_values(by=time_column, inplace=True)
                df['duration'] = df[time_column].diff().dt.total_seconds()

                if df['duration'].notna().all() and len(df) > 1:
                    kmf = KaplanMeierFitter()

                    detailed_narrative[segment] = {

def find_co_occurrences(texts, window_size, stop_words):
    all_words = []
    for text in texts:
        words = [word.lower() for word in word_tokenize(text) if word.lower() not in stop_words and word.isalpha()]
        all_words.extend(words)

    co_occurrences = []
    for i in range(len(all_words)):
        target_word = all_words[i]
        start = max(0, i - window_size)
        end = min(len(all_words), i + window_size + 1)
        for j in range(start, end):
            if i != j:
                co_occurred_word = all_words[j]
                pair = tuple(sorted([target_word, co_occurred_word]))
                co_occurrences.append(pair)

    co_occurrence_freq = FreqDist(co_occurrences)
    return co_occurrence_freq

from fpdf import FPDF
import os
import pandas as pd
import math

class PDF(FPDF):
    def __init__(self, uoi_id, user_ids, earliest_date, latest_date, total_observations):
        super().__init__(orientation='P', unit='mm', format='A4')
        self.add_font('ArialUnicodeMS', '', 'Arial_Unicode_MS_Regular.ttf', uni=True)
        self.add_font('ArialUnicodeMS', 'B', 'Arial_Unicode_MS_Regular.ttf', uni=True)
        self.add_font('ArialUnicodeMS', 'I', 'Arial_Unicode_MS_Regular.ttf', uni=True)
        self.uoi_id = uoi_id
        self.user_ids = user_ids
        self.earliest_date = earliest_date
        self.latest_date = latest_date
        self.total_observations = total_observations
        self.add_page()
        self.set_auto_page_break(auto=True, margin=15)
        self.alias_nb_pages()
        self.toc = []
        self.epw = self.w - 2 * self.l_margin  # Effective page width

    def header(self):
        self.set_font('ArialUnicodeMS', 'B', 12)
        self.set_text_color(0, 51, 102)
        self.cell(0, 10, f'Analysis Report - {self.uoi_id}', 0, 1, 'C')
        self.ln(5)

    def footer(self):
        self.set_y(-15)
        self.set_font('ArialUnicodeMS', 'I', 8)
        self.set_text_color(128)
        self.cell(0, 10, f'Page {self.page_no()}/{self.alias_nb_pages()}', 0, 0, 'C')

    def chapter_title(self, title):
        self.set_font('ArialUnicodeMS', 'B', 14)
        self.set_fill_color(200, 220, 255)
        self.cell(0, 10, title, 0, 1, 'L', 1)
        self.ln(5)
        self.toc.append((title, self.page_no()))

    def chapter_body(self, body):
        self.set_font('ArialUnicodeMS', '', 12)
        self.multi_cell(0, 10, body)
        self.ln()

    def print_image(self, image_path, width=180):
        if os.path.exists(image_path):
            self.image(image_path, w=width)
            self.ln(5)
        else:
            print(f"Image not found: {image_path}")

    def print_csv(self, csv_path, num_rows=100):
        if os.path.exists(csv_path):
            df = pd.read_csv(csv_path)
            df = df.head(num_rows)
            df = df.fillna('')
            df = df.applymap(lambda x: f'{x:.3f}' if isinstance(x, float) else x)

            self.add_page(orientation='L')
            self.set_font('ArialUnicodeMS', '', 10)
            self.set_text_color(50, 50, 50)
            col_widths = [self.epw / len(df.columns)] * len(df.columns)
            row_height = self.font_size + 2

            # Print header
            for col_name in df.columns:
                self.cell(col_widths[0], row_height, col_name, border=1)
            self.ln(row_height)

            # Print rows
            for _, row in df.iterrows():
                for i, col in enumerate(row):
                    self.multi_cell(col_widths[i], row_height, str(col), border=1)
                self.ln(row_height)

            if len(df) < num_rows:
                self.cell(0, 10, f"Showing all {len(df)} rows.", 0, 1)
            else:
                self.cell(0, 10, f"Showing first {num_rows} rows out of {len(df)}.", 0, 1)

            self.ln(5)
        else:
            print(f"CSV file not found: {csv_path}")

    def print_analysis_intro(self, analysis_type, intro_text):
        self.set_font('ArialUnicodeMS', 'B', 12)
        self.set_text_color(50, 50, 50)
        self.multi_cell(0, 10, f"{analysis_type} Analysis:")
        self.set_font('ArialUnicodeMS', '', 12)
        self.multi_cell(0, 10, intro_text)
        self.ln(5)

    def print_analysis_details(self, details):
        if details:
            self.set_font('ArialUnicodeMS', '', 12)
            self.set_text_color(100, 100, 100)
            self.multi_cell(0, 10, f"Analysis Details: {details}")
            self.ln(5)

    def print_analysis_narrative(self, narrative):
        if narrative:
            self.set_font('ArialUnicodeMS', '', 12)
            self.set_text_color(50, 50, 50)
            self.multi_cell(0, 10, f"Narrative: {narrative}")
            self.ln(5)

    def output_to_file(self, file_path):
        self.output(file_path, 'F')

    def add_toc(self):
        self.add_page()
        self.set_font('ArialUnicodeMS', 'B', 16)
        self.cell(0, 10, 'Table of Contents', 0, 1, 'L')
        self.set_font('ArialUnicodeMS', '', 12)
        for title, page_no in self.toc:
            self.cell(0, 10, f'{title} ................ {page_no}', 0, 1, 'L')
        self.ln(10)

def collect_files_by_analysis_type(directory):
    analysis_files = {}
    for file in os.listdir(directory):
        file_path = os.path.join(directory, file)
        if os.path.isfile(file_path):
            analysis_type = get_analysis_type(file)
            if analysis_type not in analysis_files:
                analysis_files[analysis_type] = []
            analysis_files[analysis_type].append(file_path)
    return analysis_files

def find_co_occurrences(texts, window_size, stop_words):
    # Tokenize and filter stopwords
    all_words = []
    for text in texts:
        words = [word.lower() for word in word_tokenize(text) if word.lower() not in stop_words and word.isalpha()]
        all_words.extend(words)

    # Find co-occurrences within the window size
    co_occurrences = []
    for i in range(len(all_words)):
        target_word = all_words[i]
        start = max(0, i - window_size)
        end = min(len(all_words), i + window_size + 1)
        for j in range(start, end):
            if i != j:
                co_occurred_word = all_words[j]
                pair = tuple(sorted([target_word, co_occurred_word]))
                co_occurrences.append(pair)

    # Frequency distribution of co-occurring word pairs
    co_occurrence_freq = FreqDist(co_occurrences)
    return co_occurrence_freq

def autonomous_survival_analysis(base_data_directory, base_results_directory, time_column, event_column):
    uoi_files = os.listdir(base_data_directory)
    for file in uoi_files:
        if file.endswith(".csv"):
            filepath = os.path.join(base_data_directory, file)
            uoi = file.split('_')[1]
            segment = '_'.join(file.split('_')[2:-1])
            uoi_dir = Path(base_results_directory) / f"UOI_{uoi}"
            uoi_dir.mkdir(parents=True, exist_ok=True)

            df = pd.read_csv(filepath)
            if time_column in df.columns and event_column in df.columns:
                df[time_column] = pd.to_datetime(df[time_column])
                df.sort_values(by=time_column, inplace=True)
                df['duration'] = df[time_column].diff().dt.total_seconds() # Time differences in seconds
                df = df.dropna(subset=['duration'])
                df['event_observed'] = df[event_column].apply(lambda x: 1 if not pd.isna(x) and x.strip() != '' else 0)

                if df['duration'].notna().all() and len(df) > 1:
                    kmf = KaplanMeierFitter()
                    kmf.fit(df['duration'], event_observed=df['event_observed'], label=segment, palette='mako')
                    kmf.plot_survival_function() # Plot survival function for the segment
                    plt.title(f'Survival Function for {segment} of UOI {uoi}')
                    plt.xlabel('Time in Seconds')
                    plt.ylabel('Survival Probability')
                    plt.savefig(uoi_dir / f'{segment}_survival_function.png')
                    plt.close()

                    console.print(f"Survival Analysis for segment '{segment}' of UOI {uoi}:")
                    console.print(f"Analyzed the time until the first violation event. The median survival time is {kmf.median_survival_time_} seconds.")
                    console.print("This analysis helps in understanding the typical duration until a policy violation occurs, providing insights into the effectiveness of content moderation over time.")
                else:
                    console.print("Insufficient data to perform survival analysis for segment '{segment}' of UOI {uoi}.")
            else:
                console.print("Required columns for survival analysis are missing in {segment} of UOI {uoi}.")
                console.print("Ensure the data includes time stamps and event observations to perform this analysis, crucial for evaluating the timing and frequency of critical events.")
